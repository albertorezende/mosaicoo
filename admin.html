Estrutura de Tabelas Supabase

-- Pixels do mosaico
create table pixels (
  id          text primary key,   -- ex: "A-10"
  row         int  not null,
  col         int  not null,
  hex_color   text not null,
  seat        text,
  event_time  timestamptz
);

-- Evento (1 linha só)
create table event (
  id          int  primary key default 1,
  event_time  timestamptz,
  grid_rows   int,
  grid_cols   int,
  active      boolean default false
);
